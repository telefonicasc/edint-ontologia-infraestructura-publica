INF01: ¿Qué equipamientos se encuentran dentro de un determinado Municipio, Distrito o Barrio? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX esadm:   <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX rdfs:    <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?facility ?facilityLabel ?adminUnit ?adminUnitLabel
WHERE {
    # Instancias de Facility
    ?facility a edintinf:Facility .

    # Vinculación con Municipio/Distrito/Barrio
    ?facility edintinf:belongsTo ?adminUnit .

    # Permitimos cualquiera de los tres tipos administrativos
    VALUES ?adminClass { esadm:Municipio esadm:Distrito esadm:Barrio }
    ?adminUnit a ?adminClass .

    # Filtrar por la unidad administrativa concreta (reemplazar por el IRI deseado)
    FILTER (?adminUnit = <URI_DIVISION_ADMIN>)

    # Etiquetas (si están disponibles)
    OPTIONAL { ?facility rdfs:label ?facilityLabel . }
    OPTIONAL { ?adminUnit rdfs:label ?adminUnitLabel . }
}
ORDER BY ?facilityLabel

SELECT ?facility ?facilityLabel ?adminUnit ?adminLabel

    # La unidad administrativa puede ser Municipio, Distrito o Barrio
    VALUES ?adminClass {
        esadm:Municipio
        esadm:Distrito
        esadm:Barrio
    }

    ?adminUnit a ?adminClass .

    # Filtro opcional para buscar por nombre (puedes eliminarlo si no lo necesitas)

    OPTIONAL { ?facility rdfs:label ?facilityLabel . }
    OPTIONAL { ?adminUnit rdfs:label ?adminLabel . }

Para equipamientos públicos 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>

SELECT ?facility ?facilityLabel ?adminUnit ?adminLabel ?owner ?ownerLabel
WHERE {
    # Equipamientos
    ?facility a edintinf:Facility ;
              edintinf:belongsTo ?adminUnit ;
              edintinf:registeredOwner ?owner .

    # La unidad administrativa puede ser Municipio, Distrito o Barrio

        esadm:Municipio

INF02: ¿Cuál es la dirección postal exacta (calle, portal) de un equipamiento específico? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX schema:   <http://schema.org/>
PREFIX esdir:    <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/direccion-postal#>
PREFIX escjr:    <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?facility ?facilityLabel ?address ?street ?streetLabel ?portal ?portalLabel
WHERE {
    # El equipamiento concreto (puedes sustituir el IRI por el que quieras)
    BIND(<URI-del-equipamiento> AS ?facility)

    OPTIONAL { ?facility rdfs:label ?facilityLabel . }

    # Dirección postal asociada a la infraestructura
    ?facility schema:address ?address .
    ?address a esdir:DireccionPostal .
    ?address escjr:contieneVia ?street .
	?address escjr:contieneVia ?portal .

    OPTIONAL { ?street rdfs:label ?streetLabel . }
    OPTIONAL { ?portal rdfs:label ?portalLabel . }
}

INF05: ¿Qué organización es responsable de la gestión de un equipamiento específico? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX skos:     <http://www.w3.org/2004/02/skos/core#>

SELECT ?facility ?facilityLabel
       ?role ?roleLabel
       ?roleType ?roleTypeLabel
       ?organization ?organizationLabel
WHERE {

    # Sustituye por el IRI del equipamiento concreto
    BIND(<URI-del-equipamiento> AS ?facility)

    OPTIONAL { ?facility rdfs:label ?facilityLabel . }

    # Rol de gestión aplicado a este equipamiento
    ?role edintinf:madeOnInfrastructure ?facility .

    OPTIONAL { ?role rdfs:label ?roleLabel . }

    # Tipo de rol (SKOS)
    ?role edintinf:hasManagementRoleType ?roleType .

    # Filtrar para que el tipo exacto sea "responsable de gestión". Revisar valores concretos de skos
    ?roleType (skos:prefLabel|rdfs:label) ?roleTypeLabel .
    FILTER(LCASE(STR(?roleTypeLabel)) = "responsable de gestión")

    # Organización que desempeña ese rol
    ?role edintinf:rolePlayedBy ?organization .
    OPTIONAL { ?organization rdfs:label ?organizationLabel . }
}
ORDER BY ?role ?organization

INF7: ¿Qué organizaciones privadas tienen un rol en infraestructuras públicas? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX trafico:  <http://vocab.ciudadesabiertas.es/def/transporte/trafico#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?infra ?infraLabel
       ?publicOwner ?publicOwnerLabel
       ?role ?roleLabel
       ?privateOrg ?privateOrgLabel
WHERE {

    # 1. Infraestructura pública
    ?infra a edintinf:InfrastructureElement ;
           edintinf:registeredOwner ?publicOwner .

    ?publicOwner trafico:isPublic true .   # Titularidad pública

    # 2. Rol de gestión sobre esa infraestructura
    ?role edintinf:madeOnInfrastructure ?infra ;
          edintinf:rolePlayedBy ?privateOrg .

    OPTIONAL { ?role rdfs:label ?roleLabel . }

    # 3. La organización que gestiona es privada
    ?privateOrg trafico:isPublic false .

    # Etiquetas opcionales
    OPTIONAL { ?infra rdfs:label ?infraLabel . }
    OPTIONAL { ?publicOwner rdfs:label ?publicOwnerLabel . }
    OPTIONAL { ?privateOrg rdfs:label ?privateOrgLabel . }
}
ORDER BY ?infra ?privateOrg

INF08: ¿Qué sensores de acceso están instalados en un equipamiento concreto? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX sosa:     <http://www.w3.org/ns/sosa/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?facility ?facilityLabel
       ?obsPoint ?obsPointLabel
       ?sensor ?sensorLabel
WHERE {

    # Sustituir por la URI del equipamiento concreto
    BIND(<URI-del-equipamiento> AS ?facility)

    OPTIONAL { ?facility rdfs:label ?facilityLabel . }

    # Puntos de observación vinculados al equipamiento
    ?facility edintinf:hasObservationPoint ?obsPoint .
    OPTIONAL { ?obsPoint rdfs:label ?obsPointLabel . }

    # Observaciones realizadas en ese punto

    ?observation sosa:madeBySensor ?sensor .

    # Filtrar solo sensores de acceso
    ?sensor a edintinf:AccessSensor .

    OPTIONAL { ?sensor rdfs:label ?sensorLabel . }
}
ORDER BY ?obsPoint ?sensor

INF09: ¿Qué observaciones de acceso de vehículos se han registrado en un intervalo de tiempo determinado? 
# INF09: ¿Qué observaciones de acceso de vehículos se han registrado en un intervalo de tiempo determinado?
También vale para INF22: ¿Cuántas observaciones de acceso de vehículos se han registrado por sensor en un intervalo de tiempo determinado? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX edintveh: <http://vocab.linkeddata.es/datosabiertos/def/transporte/censovehiculos#>
PREFIX sosa:     <http://www.w3.org/ns/sosa/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT ?observation ?observationLabel
       ?vehicle ?vehicleLabel
       ?time
       ?sensor ?sensorLabel
       ?obsPoint ?obsPointLabel
WHERE {

    # Observaciones de acceso
    ?observation a edintinf:AccessObservation .

    OPTIONAL { ?observation rdfs:label ?observationLabel . }

    # Observaciones asociadas a vehículos
    ?observation edintinf:includesVehicle ?vehicle .
    OPTIONAL { ?vehicle rdfs:label ?vehicleLabel . }

    # Sustituir por fechas de intervalo con formato: "Año-Mes-DíaTHora:Minuto:Segundo"^^xsd:dateTime
    ?observation sosa:resultTime ?time .
    FILTER ( ?time >= "FECHA_INICIO_INTERVALO"^^xsd:dateTime &&
             ?time <= "FECHA_FIN_INTERVALO"^^xsd:dateTime )

    # Sensor que realizó la observación (opcional pero útil)
    OPTIONAL { 
        ?observation sosa:madeBySensor ?sensor .
        OPTIONAL { ?sensor rdfs:label ?sensorLabel . }
    }

    # Punto de observación (opcional)
    OPTIONAL {
        ?observation sosa:hasFeatureOfInterest ?obsPoint .
        OPTIONAL { ?obsPoint rdfs:label ?obsPointLabel . }
    }
}
ORDER BY ?time ?vehicle

Para un punto de acceso concreto 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX edintveh: <http://vocab.linkeddata.es/datosabiertos/def/transporte/censovehiculos#>
PREFIX sosa:     <http://www.w3.org/ns/sosa/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT ?observation ?observationLabel
       ?vehicle ?vehicleLabel
       ?time
       ?sensor ?sensorLabel
WHERE {

    # Fijar el punto de acceso concreto
    BIND(<URI-del-punto-de-acceso> AS ?obsPoint)

    # Observaciones de acceso en ese punto
    ?observation a edintinf:AccessObservation ;
                 sosa:hasFeatureOfInterest ?obsPoint ;
                 sosa:resultTime ?time ;
                 edintinf:includesVehicle ?vehicle .

    OPTIONAL { ?observation rdfs:label ?observationLabel . }
    OPTIONAL { ?vehicle rdfs:label ?vehicleLabel . }

    # Sustituir por fechas de intervalo con formato: "Año-Mes-DíaTHora:Minuto:Segundo"^^xsd:dateTime
    FILTER ( ?time >= "FECHA_INICIO_INTERVALO"^^xsd:dateTime &&
             ?time <= "FECHA_FIN_INTERVALO"^^xsd:dateTime )

    # Sensor que realizó la observación
    OPTIONAL {
        ?observation sosa:madeBySensor ?sensor .
        OPTIONAL { ?sensor rdfs:label ?sensorLabel . }
    }
}
ORDER BY ?time ?vehicle

INF17: ¿Qué sensores están instalados en un aparcamiento concreto? 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX sosa:     <http://www.w3.org/ns/sosa/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT
       ?parking ?parkingLabel
       ?obsPoint ?obsPointLabel
       ?sensor ?sensorLabel
WHERE {

    # Sustituir por el IRI del aparcamiento concreto
    BIND(<URI-del-aparcamiento> AS ?parking)

    OPTIONAL { ?parking rdfs:label ?parkingLabel . }

    # 1. Obtener los puntos de observación del aparcamiento
    ?parking edintinf:hasObservationPoint ?obsPoint .
    OPTIONAL { ?obsPoint rdfs:label ?obsPointLabel . }

    # 2. Obtener sensores vinculados a esos puntos, a través de observaciones
    ?observation sosa:hasFeatureOfInterest ?obsPoint ;
                 sosa:madeBySensor ?sensor .

    OPTIONAL { ?sensor rdfs:label ?sensorLabel . }
}
ORDER BY ?obsPoint ?sensor
Con geolocalización de sensores 
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/infraestructura#>
PREFIX sosa:     <http://www.w3.org/ns/sosa/>
PREFIX gsp:      <http://www.opengis.net/ont/geosparql#>
PREFIX geo:      <https://datos.ign.es/def/geo_core#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT
       ?parking ?parkingLabel
       ?obsPoint ?obsPointLabel
       ?sensor ?sensorLabel
       ?geom
       ?x ?y
WHERE {

    # Aparcamiento concreto
    BIND(<URI-del-aparcamiento> AS ?parking)

    OPTIONAL { ?parking rdfs:label ?parkingLabel . }

    # Puntos de observación asociados al aparcamiento
    ?parking edintinf:hasObservationPoint ?obsPoint .
    OPTIONAL { ?obsPoint rdfs:label ?obsPointLabel . }

    # Sensores vinculados a esos puntos (vía observaciones SOSA)
    ?observation sosa:hasFeatureOfInterest ?obsPoint ;
                 sosa:madeBySensor ?sensor .

    OPTIONAL { ?sensor rdfs:label ?sensorLabel . }

    # Geometría del sensor
    ?sensor gsp:hasGeometry ?geom .

    # Coordenadas ETRS89 del sensor (X/Y)
    OPTIONAL { ?geom geo:xETRS89 ?x . }
    OPTIONAL { ?geom geo:yETRS89 ?y . }
}
ORDER BY ?obsPoint ?sensor
